166  fprintf(stderr, "layer %d is CONV layer, performs %lu multiplications \n", i, num_op);
167  num_multiplications += num_op;
168  break;
169  case FULLY_CONNECTED:
170  num_op = layer->num_inputs * layer->out_depth;
171  fprintf(stderr, "layer %d is FC layer, performs %lu multiplications \n", i, num_op);
172  num_multiplications += num_op;
173  break;
174  case POOLING:
175  num_op = layer->out_sx * layer->out_sy * layer->out_depth * layer->sx * layer->sy;
176  fprintf(stderr, "layer %d is POOL layer, performs %lu comparison \n", i, num_op);
177  num_comparison += num_op;
178  break;
179  case RELU:
180  num_op = layer->out_sx * layer->out_sy * layer->out_depth;
181  fprintf(stderr, "layer %d is RELU layer, performs %lu comparison \n", i, num_op);
182  num_comparison += num_op;
183  break;
184  case SOFTMAX:
185  fprintf(stderr, "layer %d is SMAX layer \n \n", i);
186  break;
187  default:
188  break;
189  }
